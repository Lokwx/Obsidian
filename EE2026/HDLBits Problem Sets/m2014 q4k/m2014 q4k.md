Implement the following circuit:

```
module top_module (
    input clk,
    input resetn,   // synchronous reset
    input in,
    output reg out);
    
    reg [3:0] q = {4{1'b0}};

    always @(posedge clk) begin 
        if (!resetn) q <= 0;
        else begin 
            q <= {in,q[3:1]};
        end
    end
    
    assign out = q[0];
    
endmodule
```
